    <script>
        const allEvents = [
            { event: "Magna Carta Signed", year: 1215 },
            { event: "Gutenberg Invents Printing Press", year: 1440 },
            { event: "Columbus Reaches Americas", year: 1492 },
            { event: "Start of the Protestant Reformation", year: 1517 },
            { event: "English Defeat of the Spanish Armada", year: 1588 },
            { event: "Founding of Jamestown, Virginia", year: 1607 },
            { event: "Signing of the US Declaration of Independence", year: 1776 },
            { event: "Start of the French Revolution", year: 1789 }
        ];

        const optionsContainer = document.getElementById('question-options');
        const feedbackElement = document.getElementById('feedback');
        const scoreElement = document.getElementById('score');
        
        let currentEvents = []; 
        let score = 0;
        let answered = false;
        let correctAnswerYear = 0;

        function shuffleArray(array) {
            for (let i = array.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [array[i], array[j]] = [array[j], array[i]];
            }
        }
        
        function loadNewRound() {
            answered = false;
            feedbackElement.textContent = '';
            optionsContainer.innerHTML = '';
            
            shuffleArray(allEvents);
            const selectedEvents = allEvents.slice(0, 4);

            const sortedEvents = [...selectedEvents].sort((a, b) => a.year - b.year);
            const earliestEvent = sortedEvents[0];
            correctAnswerYear = earliestEvent.year;

            shuffleArray(selectedEvents); 
            currentEvents = selectedEvents;

            currentEvents.forEach(eventData => {
                const button = document.createElement('button');
                button.className = 'event-btn';
                button.dataset.year = eventData.year; 
                button.textContent = eventData.event;
                button.addEventListener('click', checkAnswer);
                optionsContainer.appendChild(button);
            });
        }

        function checkAnswer(event) {
            if (answered) return;
            answered = true;
            
            const selectedButton = event.target;
            const selectedYear = parseInt(selectedButton.dataset.year);

            const buttons = document.querySelectorAll('.event-btn');
            buttons.forEach(btn => btn.disabled = true);

            if (selectedYear === correctAnswerYear) {
                selectedButton.classList.add('correct');
                feedbackElement.textContent = "? Correct! That event occurred first.";
                feedbackElement.className = 'correct-feedback';
                score++;
            } else {
                selectedButton.classList.add('incorrect');
                feedbackElement.textContent = `? Incorrect. The earliest event was: ${currentEvents.find(e => e.year === correctAnswerYear).event} (${correctAnswerYear}).`;
                feedbackElement.className = 'incorrect-feedback';
                
                buttons.forEach(btn => {
                    if (parseInt(btn.dataset.year) === correctAnswerYear) {
                        btn.classList.add('correct');
                    }
                });
            }

            scoreElement.textContent = `Score: ${score}`;

            setTimeout(loadNewRound, 3000); 
        }

        loadNewRound();
    </script>
